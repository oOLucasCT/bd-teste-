DROP TABLE IF EXISTS Emprestimo;
DROP TABLE IF EXISTS Livro;
DROP TABLE IF EXISTS Usuarios;

CREATE TABLE Usuarios (
	id_usuario INT PRIMARY KEY,
	nome VARCHAR(100) NOT NULL
);

CREATE TABLE Livro (
  id_livro INT PRIMARY KEY,
  titulo VARCHAR(150) NOT NULL,
  autor VARCHAR(100),
  ano INT
);

CREATE TABLE Emprestimo(
  id_emprestimo INT PRIMARY KEY,
  id_usuarios INT,
  id_livro INT,
  data_emprestimo DATE,
  data_devolucao DATE,
  FOREIGN KEY (id_usuarios) REFERENCES Usuarios(id_usuarios),
  FOREIGN KEY (id_livro) REFERENCES Livros(id_livro)
  );

DROP TABLE IF EXISTS Clientes;
DROP TABLE IF EXISTS Restaurantes;
DROP TABLE IF EXISTS Pedidos;

CREATE TABLE Clientes(
  id_clientes INTEGER PRIMARY KEY AUTOINCREMENT,
  nome VARCHAR(100) NOT NULL,
  idade DATE,
  email VARCHAR(100)
  );
  
CREATE TABLE Restaurantes(
  id_restaurantes INTEGER PRIMARY KEY AUTOINCREMENT,
  nome_restaurante VARCHAR(100) not NULL,
  nota FLOAT,
  endereco VARCHAR(100) NOT NULL
  );
  
  CREATE TABLE Entregador(
    id_entregador INTEGER PRIMARY KEY AUTOINCREMENT,
    nome_entregador TEXT,
    veiculo TEXT
    );
    
    CREATE TABLE Pagamento(
      id_pagamento INTEGER PRIMARY KEY AUTOINCREMENT,
      forma_pagamento TEXT
      );
  
  CREATE TABLE Pedidos(
    id_pedidos INTEGER PRIMARY KEY AUTOINCREMENT,
    id_clientes INT,
    id_restaurantes INT,
    qtd_pedidos INT,
    mais_pedido VARCHAR(100) not NULL,
    FOREIGN KEY (id_clientes) REFERENCES Clientes(id_clientes),
    FOREIGN Key (id_restaurantes) REFERENCES Restaurantes(id_restaurantes),
    FOREIGN KEY (id_entregador) REFERENCES Entregador(id_entreador),
    FOREIGN Key (id_pagamento) REFERENCES Pagamento(id_pagamento)
    );
